## Rest - Spread - Default Operator

```JavaScript
function f(x, ...y) { // Rest to gather
  // y is an Array
  return x * y.length;
}
f(3, "hello", true) == 6
```
```JavaScript
function f(x, y, z) {
  return x + y + z;
}
// Pass each elem of array as argument
f(...[1,2,3]) == 6 // Spread to iterate
```
```JavaScript
function f(x, y=12) { // Default to ... default
  // y is 12 if not passed (or passed as undefined)
  return x + y;
}
f(3) == 15
```
